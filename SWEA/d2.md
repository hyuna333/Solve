# 13588 구간합
```python

T = int(input())

for case in range(1, T+1):
    N, M = map(int, input().split())
    lst = list(map(int, input().split()))
    ans = []
    for i in range(N-M+1):
        add = 0
        for j in range(i, i+M):
            add += lst[j]
        ans.append(add)
    a = b = ans[0]
    for k in ans:
        if a < k:
            a = k
        elif b > k:
            b = k
    c = int(a - b)
    print(f'#{case} {c}')
```

# 13616 간단한 소인수분해
```python
T = int(input())

for case in range(1, T+1):
    N = int(input())
    a = b = c = d = e = 0
    for i in range(1, 25):
        if N % (2 ** i) == 0:
            a += 1
        if N % (3 ** i) == 0:
            b += 1
        if N % (5 ** i) == 0:
            c += 1
        if N % (7 ** i) == 0:
            d += 1
        if N % (11 ** i) == 0:
            e += 1
        lst = [a, b, c, d, e]
        result = ' '.join(map(str, lst))
    print(f'#{case} {result}')
```

# 15200. 두 개의 숫자열
```python
T = int(input())

for case in range(1, T+1):
    N, M = map(int, input().split())
    n = list(map(int, input().split()))
    m = list(map(int, input().split()))
    if N < M:
        result = []
        for i in range(M-N+1):
            lst = m[i:i+N]
            a_sum = 0
            for j in range(N):
                a = n[j] * lst[j]
                a_sum += a
            result.append(a_sum)
        # print(result)
        mx = 0
        for k in result:
            if mx < k:
                mx = k
    else:
        result = []
        for i in range(N - M + 1):
            s = n[i:i+M]
            a_sum = 0
            for j in range(M):
                a = m[j] * s[j]
                a_sum += a
            result.append(a_sum)
        mx = 0
        for k in result:
            if mx < k:
                mx = k
    print(f'#{case} {mx}')
```
