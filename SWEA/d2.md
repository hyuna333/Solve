# 1954. 달팽이 숫자
```python
T = int(input())

for case in range(1, T+1):
    N = int(input())
    lst = [[0]*N for _ in range(N)]
    # 초기위치, 방향은 0으로 둔다
    i = j = dr = 0
    # 더 할 숫자는 1부터 시작
    num = 1
    # 방향은 우 하 좌 상
    di = [0, 1, 0, -1]
    dj = [1, 0, -1, 0]

    for num in range(1, N**2+1):
        lst[i][j] = num
        ni, nj = i + di[dr], j + dj[dr]
        if 0<=ni<N and 0<=nj<N and lst[ni][nj] == 0:
            i, j = ni, nj
        else:
            dr = (dr+1)%4
            i, j = i+di[dr], j+dj[dr]
    print(f'#{case}')
    for ans in lst:
        print(*ans)
```

# 1979. 어디에 단어가 들어갈 수 있을까
```python
T = int(input())

for case in range(1, T+1):
    N, K = map(int, input().split())
    pz = [list(map(int, input().split())) for _ in range(N)]
    cnt = 0
    for i in range(N):
        dis = 0
        for j in range(N):
          # 1이면 길이에 +1
            if pz[i][j] == 1:
                dis += 1
          # 0이 나오거나 퍼즐의 맨 끝일 때 길이가 K랑 같으면 갯수에 +1 해주고 길이 초기화
            if pz[i][j] == 0 or j == N-1:
                if dis == K:
                    cnt += 1
                dis = 0
        for j in range(N):
            if pz[j][i] == 1:
                dis += 1
            if pz[j][i] == 0 or j == N - 1:
                if dis == K:
                    cnt += 1
                dis = 0
    print(f'#{case} {cnt}')
```

# 2001. 파리 퇴치
```python
T = int(input())

for case in range(1, T+1):
    N, M = map(int, input().split())
    lst = [list(map(int, input().split())) for _ in range(N)]
    mx = 0
    # i, j는 시작할 수 있는 초기 위치(왼쪽 위)
    for i in range(N-M+1):
        for j in range(N-M+1):
          # 죽인 파리의 합 세팅
            sm = 0
            # 파리채의 크기
            for h in range(M):
                for k in range(M):
                    ni = i + h
                    nj = j + k
                    sm += lst[ni][nj]
            # 최대값 찾기
            if mx < sm:
                mx = sm
    print(f'#{case} {mx}')
```

# 13666. 델타 검색
```python
T = int(input())

di = [0, 1, 0, -1]
dj = [1, 0, -1, 0]  # 우하좌상 순서


for case in range(1, T+1):
    ab_sum = 0
    N = int(input())
    arr = [list(map(int, input().split())) for _ in range(N)]
    for i in range(N):
        for j in range(N):
            for k in range(4):
                ni = i + di[k]
                nj = j + dj[k]
                if 0 <= ni < N and 0 <= nj < N:
                    bf = arr[i][j]
                    af = arr[ni][nj]
                    ab_sum += abs(bf-af)

    print(f'#{case} {ab_sum}')
```

    result = 0

                sub_set.append(lst[j])

# 13674. 색칠하기
```python
T = int(input())

for case in range(1, T+1):
    N = int(input())
    lst = [['']*10 for _ in range(10)]
    cnt = 0
    for k in range(N):
        r1, c1, r2, c2, C = map(int, input().split())
        for r in range(r1, r2+1):
            for c in range(c1, c2+1):
                if C == 1:
                    lst[r][c] += 'R'
                elif C == 2:
                    lst[r][c] += 'B'
    for i in range(len(lst)):
        for j in lst[i]:
            if 'R' in j and 'B' in j:
                cnt += 1

    print(f'#{case} {cnt}')
```
헤맨 부분 : cnt를 카운트 하는 for 구문을 for k 안에 넣어놔서 값이 잘못나왔다

# 13676. 이진탐색(반절 가르기)
```python
T = int(input())

for case in range(1, T+1):
    end, Pa, Pb = map(int, input().split())
    end_a = end_b = end
    start_a = start_b = 1
    cnt_a = cnt_b = 1
    while start_a <= end_a:
        ca = int((start_a+end_a)/2)
        if Pa == ca:
            break
        elif Pa < ca:
            end_a = ca
        else: start_a = ca
        cnt_a += 1
    while start_b <= end_b:
        cb = int((start_b+end_b)/2)
        if Pb == cb:
            break
        elif Pb < cb:
            end_b = cb
        else: start_b = cb
        cnt_b += 1
    result = ''
    if cnt_a < cnt_b: result = 'A'
    elif cnt_a > cnt_b: result = 'B'
    else: result = 0
    print(f'#{case} {result}')
```

# 13715. 회문
```python
T = int(input())

for case in range(1, T+1):
    N, M = map(int, input().split())
    lst = [input() for _ in range(N)]

    for i in range(N):
        for j in range(N - M + 1):
            mo = []
            for k in range(M):
                mo.append(lst[i][j+k])
            reword = list(reversed(mo))
            if mo == reword:
                ans = mo
                break
        for j in range(N - M + 1):
            mo = []
            for k in range(M):
                mo.append(lst[j+k][i])
            reword = list(reversed(mo))
            if mo == reword:
                ans = mo
                break
    result = ''.join(ans)
    print(f'#{case} {result}')
```
