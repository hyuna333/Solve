# 13576 gravity
```python
T = int(input())

for case in range(1, T+1):
    N = int(input())
    lst = list(map(int, input().split()))
    b = []
    for i in range(N+1):
        a = 0
        for j in range(i+1, len(lst)):
            if lst[i] > lst[j]:
                a += 1
        b.append(a)
    # print(b)
    c = b[0]
    for k in b:
        if k > c:
            c = k
    print(f'#{case} {c}')
```

T = int(input())

for test in range(1, T+1):

    a = b = lst[0]
    for i in lst:
        if a < i:
            a = i
    for j in lst:
        if b > j:
            b = j
    c = a - b
    print(f'#{test} {c}')
