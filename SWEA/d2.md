T = int(input())

for case in range(1, T+1):
    N = int(input())
    lst = list(map(int, input().split()))
    b = []
    for i in range(N+1):
        a = 0
        for j in range(i+1, len(lst)):
            if lst[i] > lst[j]:
                a += 1
        b.append(a)
    # print(b)
    c = b[0]
    for k in b:
        if k > c:
            c = k
    print(f'#{case} {c}')

# 13588 구간합
```python

T = int(input())

for case in range(1, T+1):
    N, M = map(int, input().split())
    lst = list(map(int, input().split()))
    ans = []
    for i in range(N-M+1):
        add = 0
        for j in range(i, i+M):
            add += lst[j]
        ans.append(add)
    a = b = ans[0]
    for k in ans:
        if a < k:
            a = k
        elif b > k:
            b = k
    c = int(a - b)
    print(f'#{case} {c}')
```

# 13616 간단한 소인수분해
```python
T = int(input())

for case in range(1, T+1):
    N = int(input())
    a = b = c = d = e = 0
    for i in range(1, 25):
        if N % (2 ** i) == 0:
            a += 1
        if N % (3 ** i) == 0:
            b += 1
        if N % (5 ** i) == 0:
            c += 1
        if N % (7 ** i) == 0:
            d += 1
        if N % (11 ** i) == 0:
            e += 1
        lst = [a, b, c, d, e]
        result = ' '.join(map(str, lst))
    print(f'#{case} {result}')
```
