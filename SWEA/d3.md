# 1289. 원재의 메모리 복구하기
```python
T = int(input())

for case in range(1, T+1):
    me = list(map(int, input()))
    N = len(me)
    s = [0]*N
    cnt = 0

    i = 0
    while s != me and i < N:
        # i번째 원소가 동일하면 패스
        if s[i] == me[i]:
            i += 1
        # 다르다면 변경 횟수 추가 1로 해주고 i부터 맨 끝까지 0 <> 1 바꿔주기
        else:
            cnt += 1
            for j in range(i, N):
                s[j] = (s[j]+1)%2
            i += 1

    print(f'#{case} {cnt}')
```

# 6019. 기차 사이의 파리
```python
T = int(input())

for case in range(1, T+1):
    D, A, B, F = map(int, input().split())
    move = 0
    stop = int(D / (A+B))
    for _ in range(stop):
        time1 = D / (F+B)
        move += time1 * F
        D -= (A+B) * time1
        time2 = D / (F+A)
        move += time2 * F
        D -= (A + B) * time2
    print('#{0} {1}'.format(case, round(move, 1)))
```

```python
T = 10

for case in range(1, T+1):
    n, num = map(str, input().split())
    N = int(n)
    arr = list(map(int, num))

    lst = []
    top = -1
    # 맨 뒤에서부터 확인
    for i in range(N-1, -1, -1):
      # 맨 처음 숫자거나 앞에 들어간 숫자와 다를 경우 push
        if top == -1 or arr[i] != lst[top]:
            lst.append(arr[i])
            top += 1
      # 만약 먼저 들어간 숫자와 같다면 pop으로 꺼낸다
        elif arr[i] == lst[top]:
            lst.pop(top)
            top -= 1
    lst_t = lst[::-1]

    print(f'#{case}', end=' ')
    print(*lst_t, sep='')
```

# 16497. [S/W 문제해결 기본] 6일차 - 계산기1
```python
T = 10

for case in range(1, T+1):
    N = int(input())
    lst = list(input())
    post = []
    stk1 = []
    stk2 = []
    ans = 0

    # 후위 표기식으로 바꾸기
    for i in lst:
        if i.isdigit():
            post.append(i)
        else:
            if len(stk1):
                post.append(stk1.pop())
                stk1.append(i)
            else:
                stk1.append(i)

    while stk1:
        post.append(stk1.pop())
    
    # 계산
    for i in post:
        # 숫자라면
        if i.isdigit():
            stk2.append(int(i))
        # + 라면
        else:
            if len(stk2) >= 2:
                b = stk2.pop()
                a = stk2.pop()
                stk2.append(a+b)

    if len(stk2) == 1:
        ans = stk2.pop()

    print(f'#{case} {ans}')
```
