# 13599 view
```python
T = 10

for case in range(1, T+1):
    N = int(input())
    lst = list(map(int, input().split()))
    cnt = 0
    for i in range(2, N-2):
        lar = lst[i-2]
        for j in range(i-1, i+3):
            if i == j:
                continue
            else:
                if lst[j] > lar:
                    lar = lst[j]
        if lst[i] > lar:    
            cnt += (lst[i]-lar)
    print(f'#{case} {cnt}')
```
