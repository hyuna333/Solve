# 1209. [S/W 문제해결 기본] 2일차 - Sum
```python
for case in range(1, 11):
    N = 100
    T = int(input())
    arr = [list(map(int, input().split())) for _ in range(100)]
    mx = 0
    for i in range(N):
        ga_sum = se_sum = dae_sum1 = dae_sum2 = 0
        dae_sum1 += arr[i][i]
        dae_sum2 += arr[i][N-1-i]
        for j in range(N):
            ga_sum += arr[i][j]
            se_sum += arr[j][i]
        if mx < dae_sum1: mx = dae_sum1
        if mx < dae_sum2: mx = dae_sum2
        if mx < ga_sum: mx = ga_sum
        if mx < se_sum: mx = se_sum
    print(f'#{case} {mx}')
```

# 6019. 기차 사이의 파리
```python
T = int(input())

for case in range(1, T+1):
    D, A, B, F = map(int, input().split())
    move = 0
    stop = int(D / (A+B))
    for _ in range(stop):
        time1 = D / (F+B)
        move += time1 * F
        D -= (A+B) * time1
        time2 = D / (F+A)
        move += time2 * F
        D -= (A + B) * time2
    print('#{0} {1}'.format(case, round(move, 1)))
```
왕복 생각하지말고 그냥 기차 두 대가 충돌할 때까지의 시간만 구해서 풀면되더라  
얼척이 없다
```python
T = int(input())

for case in range(1, T+1):
    D, A, B, F = map(int, input().split())
    time = D/(A+B)
    move = time * F
    print(f'#{case} {move}')
```

# 13599 view
```python
T = 10

for case in range(1, T+1):
    N = int(input())
    lst = list(map(int, input().split()))
    cnt = 0
    for i in range(2, N-2):
        lar = lst[i-2]
        for j in range(i-1, i+3):
            if i == j:
                continue
            else:
                if lst[j] > lar:
                    lar = lst[j]
        if lst[i] > lar:    
            cnt += (lst[i]-lar)
    print(f'#{case} {cnt}')
```

# 13675. 부분집합의 합
```python
T = int(input())
A = list(range(1, 13))
for case in range(1, T+1):
    N, K = map(int, input().split())
    cnt = 0
    for i in range(1, 1<<12):
        sub_sum = 0
        sub_set = []
        for j in range(12):
            if i & (1<<j):
                sub_sum += A[j]
                sub_set.append(A[j])
        if len(sub_set) == N and sub_sum == K:
            cnt += 1
    print(f'#{case} {cnt}')
```

# 13677. 특별한 정렬
```python
T = int(input())

for case in range(1, T+1):
    N = int(input())
    lst = list(map(int, input().split()))
    for i in range(N-1, -1, -1):
        for j in range(i):
            if lst[j+1] < lst[j]:
                lst[j], lst[j+1] = lst[j+1], lst[j]
    # 정렬끝
    result = [0]*10
    for k in range(0, 10, 2):
        result[k] = lst[N-1-(k//2)]
        result[k+1] = lst[k//2]
    print(f'#{case}', *result)
```

# 13998. 새로운 버스 노선
```python
T = int(input())

for case in range(1, T+1):
    stop = [0] * 1001
    N = int(input())
    for i in range(1, N+1):
        line, A, B = map(int, input().split())
        for j in range(A, B + 1):
            if line == 1:
                stop[j] += 1
            elif line == 2:
                if A % 2 and j % 2:
                    stop[j] += 1
                elif A % 2 == 0 and j % 2 == 0:
                    stop[j] += 1
            elif line == 3:
                if A % 2 and j % 3 == 0 and j % 10 != 0:
                    stop[j] += 1
                elif A % 2 == 0 and j % 4 == 0:
                    stop[j] += 1
    mx = stop[0]
    for k in stop:
        if mx < k: mx = k
    print(f'#{case} {mx}')
```
