# 13841. [S/W 문제해결 기본] 6일차 - 계산기2
```python
T = 10
dct = {'+':1, '*':2}

for case in range(1, T+1):
    N = int(input())
    lst = list(input())
    stk1 = []
    ch = []
    stk2 = []

    for i in lst:
        # 숫자면 push
        if i.isdigit():
            ch.append(i)
        # 숫자 아닐 때
        else:
            # 안에 아무것도 없거나 우선순위가 낮으면 stk push
            if len(stk1) == 0 or dct[stk1[-1]] < dct[i]:
                stk1.append(i)
            else:
                while stk1 and dct[stk1[-1]] >= dct[i]:
                    ch.append(stk1.pop())
                stk1.append(i)

    while stk1:
        ch.append(stk1.pop())

    for j in ch:
        if j.isdigit():
            stk2.append(int(j))
        else:
            if len(stk2) >= 2:
                b = stk2.pop()
                a = stk2.pop()
                if j == '+':
                    stk2.append(a+b)
                else:
                    stk2.append(a*b)

    ans = stk2.pop()

    print(f'#{case} {ans}')
```
