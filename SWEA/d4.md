# 4408. 자기 방으로 돌아가기
```python
T = int(input())

for case in range(1, T+1):
    N = int(input())
    j_lst = [0]*200

    for i in range(N):
        s, e = map(int, input().split())
        if s < e:
            sj = (s-1)//2
            ej = (e-1)//2
            for j in range(sj, ej+1):
                j_lst[j] += 1
        elif s > e:
            s, e = e, s
            sj = (s - 1) // 2
            ej = (e - 1) // 2
            for j in range(sj, ej + 1):
                j_lst[j] += 1
        else:
            j_lst[(s-1)//2] += 1

    mx = 0
    for k in j_lst:
        if mx < k:
            mx = k

    print(f'#{case} {mx}')
```

# 13841. [S/W 문제해결 기본] 6일차 - 계산기2
```python
T = 10
dct = {'+':1, '*':2}

for case in range(1, T+1):
    N = int(input())
    lst = list(input())
    stk1 = []
    ch = []
    stk2 = []

    for i in lst:
        # 숫자면 push
        if i.isdigit():
            ch.append(i)
        # 숫자 아닐 때
        else:
            # 안에 아무것도 없거나 우선순위가 낮으면 stk push
            if len(stk1) == 0 or dct[stk1[-1]] < dct[i]:
                stk1.append(i)
            else:
                while stk1 and dct[stk1[-1]] >= dct[i]:
                    ch.append(stk1.pop())
                stk1.append(i)

    while stk1:
        ch.append(stk1.pop())

    for j in ch:
        if j.isdigit():
            stk2.append(int(j))
        else:
            if len(stk2) >= 2:
                b = stk2.pop()
                a = stk2.pop()
                if j == '+':
                    stk2.append(a+b)
                else:
                    stk2.append(a*b)

    ans = stk2.pop()

    print(f'#{case} {ans}')
```

# 14020. [S/W 문제해결 기본] 9일차 - 중위순회
```python
T = 10

def tree(s):
    global ans
    if s:
        tree(L[s])
        ans += P[s]
        tree(R[s])
    
for case in range(1, T+1):
    N = int(input())
    P = ['']*(N+1)
    L = [0]*(N+1)
    R = [0]*(N+1)

    for _ in range(N):
        lst = list(input().split())
        p, A = int(lst[0]), lst[1]
        if len(lst) == 4:
            L[p] = int(lst[2])
            R[p] = int(lst[3])
        elif len(lst) == 3:
            L[p] = int(lst[2])
        P[p] = A

    ans = ''

    tree(1)

    print(f'#{case} {ans}')
```

# 14032. 정사각형 방
```python
T = int(input())

def move(a, b):
  # 시작점은 고정해야하므로 global 변수 사용
    global si, sj, room
    ci, cj = a, b
    for di, dj in ((-1, 0), (0, 1), (1, 0), (0, -1)):
        ni, nj = ci + di, cj + dj
        # 범위내이고 1 차이 날 경우에 이동+1 하고 좌표를 바꿔준다
        if 0 <= ni < N and 0 <= nj < N and arr[ci][cj] + 1 == arr[ni][nj]:
            ci, cj = ni, nj
            room[arr[si][sj]] += 1
            move(ci, cj)
            break

for case in range(1, T+1):
    N = int(input())
    arr = []
    room = [0]*(N**2+1)

    for _ in range(N):
        arr.append(list(map(int, input().split())))

    for si in range(N):
        for sj in range(N):
            move(si, sj)

    mx = 0
    mx_idx = 0
    for i in range(1, len(room)):
        if mx < room[i]:
            mx = room[i]
            mx_idx = i

    print(f'#{case} {mx_idx} {mx+1}')
```
